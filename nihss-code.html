// Show text field when UN is selected
Document.querySelectorAll(input[value=UN]).forEach(radio => {
Radio.addEventListener(change, function() {
Const noteField = document.getElementById(this.name + -
note);
If (this.checked) {
noteField.style.display = block;
} else {
noteField.style.display = none;
}
});
});
// Hide all note fields initially
Document.querySelectorAll([id$=-note]).forEach(note => {
Note.style.display = none;
});
// Calculate NIHSS total score
Document.getElementById(calculate-btn).addEventListener(click,
function() {
Let total = 0;
Const scoreItems = [
item1a, item1b, item1c, item2, item3, item4,
item5a, item5b, item6a, item6b, item7, item8,
item9, item10, item11
];
scoreItems.forEach(item => {
const selected =
document.querySelector(`input[name=${item}]:checked`);
if (selected && selected.value !== UN) {
total += parseInt(selected.value);
}
});
Document.getElementById(total-score).textContent = total;
});
// Export as PDF
Document.getElementById(export-pdf).addEventListener(click,
function() {
// First calculate the score
Document.getElementById(calculate-btn).click();
// Prepare for printing
Window.print();
});
// Auto-fill date
Window.onload = function() {
Const today = new Date();
Const yyyy = today.getFullYear();
Const mm = String(today.getMonth() + 1).padStart(2, 0);
Const dd = String(today.getDate()).padStart(2, 0);
Document.getElementById(date).value = `${yyyy}-${mm}-
${dd}`;
Const hours = String(today.getHours()).padStart(2, 0);
Const minutes = String(today.getMinutes()).padStart(2, 0);
Document.getElementById(time).value = `${hours}:${minutes}`;
};
